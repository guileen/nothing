 var a = document.getElementById("foo");
 /(^|\s)fee(\s|$)/.test(a.className) ?
 a.className = a.className.replace(/(^|\s)fee(\s|$)/, " ") :
 /(^|\s)fee(\s|$)/.test(a.className) || (a.className += " fee");
